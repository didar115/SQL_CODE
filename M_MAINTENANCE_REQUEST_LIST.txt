SELECT ROWNUM sl,
 (SELECT NAME FROM MDA.MW_VESSELS WHERE ID  =  R.VESSEL_ID) VESSEL,
       D.NAME DEPARTMENT,
       (SELECT E.EMP_NAME
          FROM PMIS.EMPLOYEE E
         WHERE R.VESSEL_MASTER_ID = E.EMP_ID)
          VESSEL_MASTER,
       R.DESCRIPTION,
       S.NAME STATUS,
       (SELECT E.EMP_NAME
          FROM PMIS.EMPLOYEE E
         WHERE R.REQUESTER_EMP_ID = E.EMP_ID)
          REQUESTER_EMP, R.CREATED_AT REQUEST_DATE
  FROM MDA.MW_MAINTENANCE_REQS R,  MDA.MW_STATUS_CONFIG S,  MDA.MW_DEPARTMENTS D
 WHERE R.STATUS = S.STATUS_CODE AND R.DEPARTMENT_ID = D.ID
 
 and S.STATUS_CODE = nvl(:p_STATUS_CODE , s.status_code)
 and R.VESSEL_ID =nvl(:p_vessel_id , r.vessel_id)
 and R.CREATED_AT between nvl(:p_start_date , r.created_at) and nvl (:p_end_date , r.created_at)
